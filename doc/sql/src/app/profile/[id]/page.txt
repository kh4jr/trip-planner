#### [SECTION 2] Database Architectural Review & CRUD Mapping
Plik `page.tsx` (w katalogu `profile/[id]`) to komponent serwerowy Next.js renderujący publiczny profil innego użytkownika. Dostęp do danych profilowych jest ściśle chroniony przez bazę danych i wymaga potwierdzenia relacji znajomości o statusie `"ACCEPTED"`.

Identyfikacja operacji CRUD:
- **Select (Odczyt)**:
  - `db.user.findUnique` – pobranie podstawowych informacji o profilu wybranego użytkownika (klucz główny `id`).
  - `db.friend.findFirst` – weryfikacja istnienia relacji przyjaźni o statusie `"ACCEPTED"` między zalogowanym użytkownikiem a właścicielem profilu.
  - `db.trip.findMany` – pobranie listy podróży należących do wybranego użytkownika wraz z ich kosztami (`expenses`) w celu obliczenia statystyk podróżniczych (budżet, kraje).

Interpretacja SQL:
1. **Wyszukiwanie użytkownika (`db.user.findUnique`)**:
   ```sql
   SELECT "id", "email", "name" 
   FROM "User" 
   WHERE "id" = $1;
   ```
2. **Weryfikacja relacji znajomości (`db.friend.findFirst`)**: Złożony warunek logiczny mapujący relację dwukierunkową:
   ```sql
   SELECT "id", "userId", "friendId", "status" 
   FROM "Friend" 
   WHERE "status" = 'ACCEPTED' 
     AND (
       ("userId" = $1 AND "friendId" = $2) 
       OR 
       ("userId" = $3 AND "friendId" = $4)
     ) 
   LIMIT 1;
   ```
   Zapytanie to jest optymalizowane przez unikalny indeks złożony `@@unique([userId, friendId])` zadeklarowany w schemacie Prisma.
3. **Pobieranie wycieczek i wydatków (`db.trip.findMany`)**:
   ```sql
   SELECT "id", "name", "destination", "startDate", "endDate", "ownerId", "description", "location", "createdAt" 
   FROM "Trip" 
   WHERE "ownerId" = $1 
   ORDER BY "startDate" DESC;
   
   -- Pobranie powiązanych wydatków
   SELECT * FROM "Expense" WHERE "tripId" IN ($1, $2, ...);
   ```

#### [SECTION 3] Code Review & Explanatory Comments
```typescript
import React from "react";
import Link from "next/link";
import { getServerSession } from "next-auth";
import { authOptions } from "@/app/api/auth/[...nextauth]/route";
import { db } from "@/lib/db";
import { redirect, notFound } from "next/navigation";
import WorldMap from "@/components/WorldMap";

interface PageProps {
  params: {
    id: string; // Identyfikator użytkownika, którego profil jest przeglądany.
  };
}

export default async function FriendProfilePage({ params }: PageProps) {
  // Pobranie bieżącej sesji w celu autoryzacji żądania.
  const session = await getServerSession(authOptions);

  // Jeśli brak sesji (użytkownik niezalogowany), następuje przekierowanie na stronę główną.
  if (!session?.user?.id) {
    redirect("/");
  }

  const viewerId = Number(session.user.id);
  const profileUserId = Number(params.id);

  // W przypadku próby wejścia na własny profil za pomocą identyfikatora w URL, następuje przekierowanie na profil osobisty.
  if (viewerId === profileUserId) {
    redirect("/profile");
  }

  // Odczyt z bazy danych profilu użytkownika na podstawie klucza głównego (id).
  // Działa szybko dzięki indeksowi PK na tabeli User.
  const user = await db.user.findUnique({
    where: { id: profileUserId },
  });

  // Brak profilu o danym ID zwraca błąd 404.
  if (!user) {
    notFound();
  }

  // Krytyczne sprawdzenie autoryzacji: weryfikacja, czy przeglądający (viewerId) 
  // posiada zaakceptowaną znajomość (status: ACCEPTED) z właścicielem profilu (profileUserId).
  // Wykorzystuje dwukierunkowy warunek OR dla pary kluczy obcych.
  const friendship = await db.friend.findFirst({
    where: {
      status: "ACCEPTED",
      OR: [
        { userId: viewerId, friendId: profileUserId },
        { userId: profileUserId, friendId: viewerId },
      ],
    },
  });

  // Profil jest widoczny wyłącznie dla zaakceptowanych znajomych.
  if (!friendship) {
    notFound(); 
  }

  // Pobranie z bazy danych wszystkich podróży należących do znajomego wraz z powiązanymi wydatkami.
  // Dane posortowane są malejąco po dacie rozpoczęcia ("startDate": "desc").
  const userTrips = await db.trip.findMany({
    where: {
      ownerId: profileUserId,
    },
    include: {
      expenses: true, // Eager Loading wydatków w celu agregacji sumy kosztów wyjazdu.
    },
    orderBy: {
      startDate: "desc",
    },
  });

  // Obliczenia statystyk na podstawie pobranych danych z bazy (wykonywane w pamięci RAM aplikacji).
  const totalTrips = userTrips.length;

  // Obliczenie sumy wydatków ze wszystkich podróży.
  const totalExpensesSum = userTrips.reduce((acc, trip) => {
    const tripSum =
      trip.expenses?.reduce((s, e) => {
        const val = Number(e.amount?.toString() || 0);
        return s + (isNaN(val) ? 0 : val);
      }, 0) || 0;

    return acc + tripSum;
  }, 0);

  // Liczba wyjazdów zakończonych (data końcowa mniejsza niż czas obecny).
  const completedTrips = userTrips.filter(
    (t) => t.endDate && new Date(t.endDate) < new Date()
  ).length;

  // Liczba unikalnych celów podróży (zastosowanie struktury Set eliminuje duplikaty lokalizacji).
  const uniqueDestinations = new Set(
    userTrips.map((t) => t.destination).filter(Boolean)
  ).size;

  const visitedCountries = Array.from(
    new Set(userTrips.map((t) => t.destination).filter(Boolean))
  ) as string[];

  return (
    <div className="!w-full min-h-screen bg-[#F8FAFC] !p-6 md:!p-12 text-left">
      <Link
        href="/"
        className="flex items-center gap-2 text-blue-400 font-black text-xs uppercase tracking-widest mb-8 hover:text-blue-600 transition-all w-fit"
      >
        <span className="text-xl">←</span> Powrót do planowania
      </Link>

      <div className="max-w-6xl !mx-0 flex flex-col gap-8">
        {/* KARTA PROFILU */}
        <section className="bg-white rounded-[3rem] p-10 shadow-2xl shadow-blue-100/50 border border-blue-50 flex flex-col md:flex-row items-center gap-8">
          <div className="w-32 h-32 bg-blue-600 rounded-[2rem] flex items-center justify-center text-white text-5xl font-black shadow-xl border-4 border-blue-50">
            {user.name?.charAt(0).toUpperCase()}
          </div>
          <div className="text-center md:text-left">
            <h2 className="text-4xl font-black text-blue-900 mb-2">
              {user.name}
            </h2>
            <p className="text-blue-400 font-bold">Znajomy podróżnik</p>
            <div className="flex gap-3 mt-4 justify-center md:justify-start">
              <span className="bg-blue-100 text-blue-600 px-4 py-1.5 rounded-xl text-xs font-black uppercase">
                ✈️ {totalTrips} Wypraw
              </span>
            </div>
          </div>
        </section>

        <div className="grid grid-cols-1 lg:grid-cols-3 gap-8">
          {/* STATYSTYKI */}
          <div className="lg:col-span-1 grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-1 gap-6">
            <StatCard title="Wyjazdy" value={totalTrips} sub="Łącznie" icon="🗺️" />
            <StatCard
              title="Lokalizacje"
              value={uniqueDestinations}
              sub="Odwiedzone"
              icon="🌍"
            />
            <StatCard
              title="Budżet"
              value={`${totalExpensesSum.toFixed(2)} zł`}
              sub="Łącznie"
              icon="💰"
            />
            <StatCard
              title="Zakończone"
              value={completedTrips}
              sub="Wyjazdy"
              icon="📅"
            />
          </div>

          {/* MAPA */}
          <div className="lg:col-span-2 bg-white rounded-[3rem] p-10 shadow-xl border border-blue-50">
            <h3 className="text-2xl font-black text-blue-900 mb-6">
              Mapa Odkryć
            </h3>
            <div className="w-full min-h-[300px] bg-blue-50/30 rounded-[2.5rem] border-2 border-dashed border-blue-100 flex items-center justify-center overflow-hidden">
              <WorldMap visited={visitedCountries} />
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}

// Komponent pomocniczy renderujący kartę pojedynczej statystyki podróżniczej.
function StatCard({
  title,
  value,
  sub,
  icon,
}: {
  title: string;
  value: string | number;
  sub: string;
  icon: string;
}) {
  return (
    <div className="bg-white p-8 rounded-[2.5rem] shadow-xl border border-blue-50 text-left">
      <div className="text-3xl mb-4">{icon}</div>
      <p className="text-[10px] font-black text-blue-300 uppercase tracking-widest mb-1">
        {title}
      </p>
      <h3 className="text-3xl font-black text-blue-900">{value}</h3>
      <p className="text-[10px] text-slate-400 mt-1 font-bold">{sub}</p>
    </div>
  );
}
```
